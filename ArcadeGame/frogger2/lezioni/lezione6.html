Lezione 6: Collisioni

In questa lezione approfondiremo le tecniche per rilevare le collisioni tra oggetti di gioco, miglioreremo il sistema di rilevamento già implementato e aggiungeremo effetti visivi per rendere più chiare le interazioni tra gli elementi.

1. Rilevamento delle collisioni

Il rilevamento delle collisioni è fondamentale in qualsiasi videogioco. Esistono diverse tecniche per implementarlo:

  • Collisione per punto: verifica se un punto si trova all'interno di un'area
  • Collisione AABB (Axis-Aligned Bounding Box): verifica se due rettangoli si sovrappongono
  • Collisione per cerchi: verifica se la distanza tra i centri di due cerchi è minore della somma dei loro raggi
  • Collisione per pixel: verifica la sovrapposizione esatta dei pixel (più complessa e costosa)

2. Implementazione di collisioni precise in Frogger

Nel nostro gioco utilizzeremo principalmente la collisione AABB, che è adatta alla struttura a griglia di Frogger:

// Funzione per verificare se due rettangoli si sovrappongono
function checkAABBCollision(rect1, rect2) {
    return (
        rect1.x < rect2.x + rect2.width &&
        rect1.x + rect1.width > rect2.x &&
        rect1.y < rect2.y + rect2.height &&
        rect1.y + rect1.height > rect2.y
    );
}

// Esempio di utilizzo
const frogRect = { x: frogX, y: frogY, width: 1, height: 1 };
const carRect = { x: car.x, y: car.y, width: car.width, height: 1 };
if (checkAABBCollision(frogRect, carRect)) {
    console.log('Collisione avvenuta!');
}

        // Stato del gioco
        let frogX = 5;

        let lives = 3;
        let gameTime = 0;

        // Definizione degli ostacoli

            { x: 0, y: 7, width: 3, height: 0.8, speed: 0.05, type: 'truck', color: 'orange' },

            { x: 0, y: 10, width: 2, height: 0.8, speed: 0.04, type: 'car', color: 'green' },
            { x: 5, y: 11, width: 3, height: 0.8, speed: -0.02, type: 'truck', color: 'purple' },

        // Funzione per verificare la collisione AABB tra due rettangoli
        function checkAABBCollision(rect1, rect2) {
            return (
                rect1.x < rect2.x + rect2.width &&
                rect1.x + rect1.width > rect2.x &&
                rect1.y < rect2.y + rect2.height &&
                rect1.y + rect1.height > rect2.y
            );
        }
        
        // Funzione per disegnare la griglia
        function drawGrid() {
            for (let y = 0; y < rows; y++) {
                for (let x = 0; x < columns; x++) {
                    switch(gameLayout[y][x]) {
                        case 'W': ctx.fillStyle = '#000080'; break;
                        case 'R': ctx.fillStyle = '#505050'; break;
                        case 'S': ctx.fillStyle = '#0f5f0f'; break;
                    }
                    ctx.fillRect(x * gridSize, y * gridSize, gridSize, gridSize);
                    ctx.strokeStyle = '#333';
                    ctx.strokeRect(x * gridSize, y * gridSize, gridSize, gridSize);
                }
            }
            
            // Disegna il traguardo
            for (let x = 0; x < columns; x++) {
                if (x % 2 === 0) {
                    ctx.fillStyle = '#FFD700'; // Oro
                    ctx.fillRect(x * gridSize, 0, gridSize, gridSize/3);
                }
            }
        }
        
        // Funzione per disegnare gli ostacoli
        function drawObstacles() {
            for (const obs of obstacles) {
                // Colore base dell'ostacolo
                ctx.fillStyle = obs.color;
                
                // Disegna l'ostacolo
                ctx.fillRect(
                    obs.x * gridSize,
                    obs.y * gridSize + gridSize * (1-obs.height) / 2,
                    obs.width * gridSize,
                    obs.height * gridSize
                );
                
                // Aggiungi dettagli in base al tipo
                if (obs.type === 'car') {
                    // Finestrini
                    ctx.fillStyle = '#87CEEB';
                    ctx.fillRect(
                        (obs.speed > 0 ? obs.x + 0.3 : obs.x + obs.width - 1.3) * gridSize,
                        obs.y * gridSize + gridSize * 0.25,
                        gridSize * 0.6,
                        gridSize * 0.3
                    );
                    
                    // Ruote
                    ctx.fillStyle = 'black';
                    ctx.fillRect(
                        obs.x * gridSize + gridSize * 0.2,
                        obs.y * gridSize + gridSize * 0.7,
                        gridSize * 0.3,
                        gridSize * 0.2
                    );
                    ctx.fillRect(
                        (obs.x + obs.width - 0.5) * gridSize,
                        obs.y * gridSize + gridSize * 0.7,
                        gridSize * 0.3,
                        gridSize * 0.2
                    );
                } else if (obs.type === 'truck') {
                    // Cabina
                    ctx.fillStyle = '#A9A9A9';
                    ctx.fillRect(
                        (obs.speed > 0 ? obs.x : obs.x + obs.width - 0.8) * gridSize,
                        obs.y * gridSize + gridSize * 0.15,
                        gridSize * 0.8,
                        gridSize * 0.5
                    );
                    
                    // Ruote
                    ctx.fillStyle = 'black';
                    for (let i = 0; i < obs.width; i++) {
                        ctx.fillRect(
                            (obs.x + i + 0.3) * gridSize,
                            obs.y * gridSize + gridSize * 0.7,
                            gridSize * 0.4,
                            gridSize * 0.2
                        );
                    }
                } else if (obs.type === 'log') {
                    // Texture del tronco (linee)
                    ctx.strokeStyle = '#5D4037';
                    ctx.lineWidth = 2;
                    for (let i = 0; i < obs.width; i++) {
                        ctx.beginPath();
                        ctx.moveTo((obs.x + i) * gridSize + 10, obs.y * gridSize + gridSize * 0.3);
                        ctx.lineTo((obs.x + i) * gridSize + gridSize - 10, obs.y * gridSize + gridSize * 0.3);
                        ctx.stroke();
                        
                        ctx.beginPath();
                        ctx.moveTo((obs.x + i) * gridSize + 5, obs.y * gridSize + gridSize * 0.7);
                        ctx.lineTo((obs.x + i) * gridSize + gridSize - 5, obs.y * gridSize + gridSize * 0.7);
                        ctx.stroke();
                    }
                }
                
                // Debug: mostra hitbox se attivato
                if (document.getElementById('debugCollisions').checked) {
                    ctx.strokeStyle = 'red';
                    ctx.lineWidth = 2;
                    ctx.strokeRect(
                        obs.x * gridSize,
                        obs.y * gridSize,
                        obs.width * gridSize,
                        gridSize
                    );
                }
            }
        }
        
        // Funzione per disegnare la rana
        function drawFrog() {
            // Colore base della rana (cambia se colpita)
            if (hitEffect > 0) {
                // Effetto lampeggiante rosso quando colpita
                ctx.fillStyle = hitEffect % 2 === 0 ? 'red' : 'white';
                hitEffect--;
            } else if (winEffect > 0) {
                // Effetto lampeggiante oro quando vince
                ctx.fillStyle = winEffect % 2 === 0 ? '#FFD700' : 'lime';
                winEffect--;
            } else {
                ctx.fillStyle = 'lime';
            }
            
            // Corpo della rana (rettangolo principale)
            ctx.fillRect(
                frogX * gridSize + gridSize/4,
                frogY * gridSize + gridSize/4,
                gridSize/2,
                gridSize/2
            );
            
            // Occhi (solo se non in stato di hit)
            if (hitEffect === 0) {
                ctx.fillStyle = 'black';
                ctx.beginPath();
                ctx.arc(
                    frogX * gridSize + gridSize/3, 
                    frogY * gridSize + gridSize/3, 
                    3, 0, Math.PI * 2
                );
                ctx.arc(
                    frogX * gridSize + 2*gridSize/3, 
                    frogY * gridSize + gridSize/3, 
                    3, 0, Math.PI * 2
                );
                ctx.fill();
            }
            
            // Debug: mostra hitbox se attivato
            if (document.getElementById('debugCollisions').checked) {
                ctx.strokeStyle = 'red';
                ctx.lineWidth = 2;
                ctx.strokeRect(
                    (frogX + 0.25) * gridSize,
                    (frogY + 0.25) * gridSize,
                    gridSize * 0.5,
                    gridSize * 0.5
                );
            }
        }
        
        // Funzione per aggiornare le posizioni degli ostacoli
        function updateObstacles(deltaTime) {
            for (const obs of obstacles) {
                // Aggiorna la posizione in base alla velocità
                obs.x += obs.speed * deltaTime;
                
                // Wrapping degli ostacoli quando escono dallo schermo
                if (obs.speed > 0 && obs.x > columns) {
                    obs.x = -obs.width;
                } else if (obs.speed < 0 && obs.x < -obs.width) {
                    obs.x = columns;
                }
                
                // Se la rana è su un tronco, si muove con esso
                if (obs.type === 'log' && frogY === obs.y) {
                    // Verifica se la rana è effettivamente sul tronco usando AABB
                    const frogRect = {
                        x: frogX + 0.25,
                        y: frogY + 0.25,
                        width: 0.5,
                        height: 0.5
                    };
                    
                    const logRect = {
                        x: obs.x,
                        y: obs.y,
                        width: obs.width,
                        height: 1
                    };
                    
                    if (checkAABBCollision(frogRect, logRect)) {
                        frogX += obs.speed * deltaTime;
                    }
                }
            }
        }
        
        // Funzione per controllare le collisioni
        function checkCollisions() {
            // Controlla se la rana è uscita dai bordi
            if (frogX < 0) frogX = 0;
            if (frogX > columns - 1) frogX = columns - 1;
            
            // Crea il rettangolo di collisione della rana
            const frogRect = {
                x: frogX + 0.25, // Riduciamo leggermente l'hitbox per collisioni più precise
                y: frogY + 0.25,
                width: 0.5,
                height: 0.5
            };
            
            // Verifica se la rana è nell'acqua
            if (frogY >= 1 && frogY <= 3) {
                let onLog = false;
                
                for (const obs of obstacles) {
                    if (obs.type === 'log') {
                        const logRect = {
                            x: obs.x,
                            y: obs.y,
                            width: obs.width,
                            height: 1
                        };
                        
                        if (checkAABBCollision(frogRect, logRect)) {
                            onLog = true;
                            break;
                        }
                    }
                }
                
                if (!onLog) {
                    die('SPLASH!');
                }
            }
            
            // Verifica collisioni con veicoli
            if ((frogY >= 5 && frogY <= 8) || (frogY >= 10 && frogY <= 11)) {
                for (const obs of obstacles) {
                    if (obs.type === 'car' || obs.type === 'truck') {
                        const vehicleRect = {
                            x: obs.x,
                            y: obs.y,
                            width: obs.width,
                            height: 1
                        };
                        
                        if (checkAABBCollision(frogRect, vehicleRect)) {
                            die('SPLAT!');
                            break;
                        }
                    }
                }
            }
            
            // Verifica se la rana ha raggiunto il traguardo
            if (frogY === 0) {
                win();
            }
        }
        
        // Funzione per la morte della rana
        function die(message) {
            if (hitEffect > 0) return; // Se già in stato di morte, ignora
            
            lives--;
            document.getElementById('lives').textContent = lives;
            hitEffect = 10; // Imposta l'effetto visivo (10 frames di lampeggio)
            
            // Mostra messaggio
            ctx.fillStyle = 'red';
            ctx.font = '30px Arial';
            ctx.fillText(message, canvas.width/2-60, canvas.height/2);
            
            if (lives <= 0) {
                // Game over
                setTimeout(() => {
                    alert('Game Over! Il gioco ricomincerà.');
                    resetGame();
                }, 1000);
            } else {
                // Resetta la posizione della rana dopo un secondo
                setTimeout(() => {
                    frogX = 5;
                    frogY = 12;
                }, 1000);
            }
        }
        
        // Funzione per la vittoria
        function win() {
            winEffect = 10; // Effetto vittoria
            
            // Mostra messaggio
            ctx.fillStyle = 'yellow';
            ctx.font = '30px Arial';
            ctx.fillText('HAI VINTO!', canvas.width/2-80, canvas.height/2);
            
            // Resetta la posizione della rana dopo 1.5 secondi
            setTimeout(() => {
                frogX = 5;
                frogY = 12;
                // Aumenta velocità degli ostacoli per incrementare difficoltà
                obstacles.forEach(obs => obs.speed *= 1.1);
            }, 1500);
        }
        
        // Reset del gioco
        function resetGame() {
            lives = 3;
            gameTime = 0;
            frogX = 5;
            frogY = 12;
            document.getElementById('lives').textContent = lives;
            document.getElementById('time').textContent = '0';
            
            // Ripristina velocità originali degli ostacoli
            obstacles.forEach((obs, index) => {
                if (index === 0) obs.speed = 0.03;
                else if (index === 1) obs.speed = -0.04;
                else if (index === 2) obs.speed = 0.05;
                else if (index === 3) obs.speed = -0.03;
                else if (index === 4) obs.speed = 0.04;
                else if (index === 5) obs.speed = -0.02;
                else if (index === 6) obs.speed = 0.02;
                else if (index === 7) obs.speed = -0.03;
                else if (index === 8) obs.speed = 0.025;
            });
        }
        
        // Funzione per muovere la rana
        function moveFrog(direction) {
            if (!canMove || paused || hitEffect > 0) return;
            
            const prevX = frogX;
            const prevY = frogY;
            
            switch(direction) {
                case 'up': if (frogY > 0) frogY -= 1; break;
                case 'down': if (frogY < rows - 1) frogY += 1; break;
                case 'left': if (frogX > 0) frogX -= 1; break;
                case 'right': if (frogX < columns - 1) frogX += 1; break;
            }
            
            if (prevX !== frogX || prevY !== frogY) {
                canMove = false;
                setTimeout(() => {
                    canMove = true;
                    checkInput();
                }, moveDelay);
            }
            
            // Se è il primo movimento, il gioco inizia
            if (!gameStarted) {
                gameStarted = true;
            }
        }
        
        // Funzione per controllare l'input
        function checkInput() {
            if (!canMove || paused || hitEffect > 0) return;
            
            if (keys.ArrowUp) moveFrog('up');
            else if (keys.ArrowDown) moveFrog('down');
            else if (keys.ArrowLeft) moveFrog('left');
            else if (keys.ArrowRight) moveFrog('right');
        }
        
        // Funzione principale di disegno
        function draw() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            drawGrid();
            drawObstacles();
            drawFrog();
            
            // Se non abbiamo ancora iniziato, mostriamo un messaggio
            if (!gameStarted) {
                ctx.fillStyle = 'rgba(0, 0, 0, 0.7)';
                ctx.fillRect(0, 0, canvas.width, canvas.height);
                
                ctx.fillStyle = 'white';
                ctx.font = '30px Arial';
                ctx.fillText('Premi una freccia per iniziare', canvas.width/2-180, canvas.height/2);
            }
        }
        
        // Loop principale del gioco
        function update(timestamp) {
            // Calcola delta time
            const deltaTime = (timestamp - lastTime) / 1000; // in secondi
            lastTime = timestamp;
            
            // Aggiorna tempo di gioco
            if (gameStarted && !paused && hitEffect === 0) {
                gameTime += deltaTime;
                document.getElementById('time').textContent = Math.floor(gameTime);
            }
            
            // Se il gioco è in corso, aggiorna gli elementi
            if (gameStarted && !paused && hitEffect === 0) {
                updateObstacles(deltaTime);
                checkCollisions();
            }
            
            // Disegna lo stato corrente
            draw();
            
            // Continua il loop
            requestAnimationFrame(update);
        }
        
        // Event listeners per la tastiera
        document.addEventListener('keydown', (event) => {
            if (event.key in keys) {
                keys[event.key] = true;
                event.preventDefault(); // Previene lo scrolling della pagina
                
                // Avvia il gioco al primo input
                if (!gameStarted) {
                    gameStarted = true;
                }
                
                checkInput();
            }
        });
        
        document.addEventListener('keyup', (event) => {
            if (event.key in keys) {
                keys[event.key] = false;
                event.preventDefault();
            }
        });
        
        // Event listeners per i bottoni touch
        document.getElementById('btnUp').addEventListener('click', () => {
            if (!gameStarted) gameStarted = true;
            moveFrog('up');
        });
        document.getElementById('btnDown').addEventListener('click', () => {
            if (!gameStarted) gameStarted = true;
            moveFrog('down');
        });
        document.getElementById('btnLeft').addEventListener('click', () => {
            if (!gameStarted) gameStarted = true;
            moveFrog('left');
        });
        document.getElementById('btnRight').addEventListener('click', () => {
            if (!gameStarted) gameStarted = true;
            moveFrog('right');
        });
        
        // Toggle di pausa
        document.getElementById('btnPause').addEventListener('click', () => {
            if (gameStarted) {
                paused = !paused;
                document.getElementById('btnPause').textContent = paused ? 'Riprendi' : 'Pausa';
            }
        });
        
        // Avvia il loop di gioco
        requestAnimationFrame(update);
    </script>
</body>

4. Spiegazione del codice

Sistema di collisioni AABB

Abbiamo implementato un sistema di collisioni basato su rettangoli allineati agli assi (AABB). La funzione checkAABBCollision verifica se due rettangoli si sovrappongono, considerando posizione, larghezza e altezza.

Hitbox e debug visivo

Abbiamo aggiunto:

  • Hitbox più precise per la rana e gli ostacoli
  • Una modalità debug per visualizzare i rettangoli di collisione
  • Effetti visivi quando si verificano collisioni (lampeggiamento)

Sistema di vite

È stato implementato un sistema di vite che:

  • Inizia con 3 vite
  • Diminuisce quando la rana viene colpita o finisce nell'acqua
  • Termina il gioco quando arrivano a zero
